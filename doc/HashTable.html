def search(key)
        index = hash_function(key)
        if @table[index] == nil
                return -1
        else
                slot_num = 1 
                @table[index].each do |slot|
                                
                        if slot[:key] == key
                                return {:location => "#{index + 1}/#{slot_num}", :data => slot[:value]}
                        end
                        slot_num += 1 
                end
                return -1
        end
end
