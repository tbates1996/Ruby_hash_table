def search_from_file
        print_search_header
        search_keys = File.readlines("SEARCH.txt")
        search_keys.each do |key|
                result = search(key.chomp)
                if result == -1
                        print_search(key.chomp, "Record not found" , "")
                else
                        print_search(key.chomp, result[:data] , result[:location])
                end
        end
end
